// Baris ini harus ada di atas agar fungsi tahu mau kirim data ke mana const REPO = "ikyshinicy/nama-repo-tuan"; async function pushUpdate() { const token = localStorage.getItem('iky_token'); if(!token) return alert("Maaf Tuan, Token belum ditanam di localStorage!"); const body = { name: document.getElementById('db-name').value, bio: document.getElementById('db-bio').value, projects: [] }; const btn = document.querySelector('.btn[onclick="pushUpdate()"]'); if(btn) { btn.innerHTML = ' PUSHING...'; btn.disabled = true; } try { const getRes = await fetch(`https://api.github.com/repos/${REPO}/contents/data.json`, { headers: { 'Authorization': `token ${token}` } }); const currentData = await getRes.json(); const latestSha = currentData.sha; const res = await fetch(`https://api.github.com/repos/${REPO}/contents/data.json`, { method: 'PUT', headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: "Update via Core Engine - Tuan Iky", content: btoa(unescape(encodeURIComponent(JSON.stringify(body, null, 2)))), sha: latestSha }) }); if(res.ok) { alert("✅ UPDATE BERHASIL!"); location.reload(); } else { alert("❌ GAGAL PUSH: Pastikan Repo & Token Benar"); } } catch (err) { alert("⚠️ ERROR: " + err.message); } finally { if(btn) btn.disabled = false; } }
